Projeto Elevador

Projetos-Códigos

Consiste basicamente em um simulador de elevador

Como em um elevador normal, ele tem algumas regras lógicas, não podendo trocar de andar caso as portas estiverem abertas, e nem se tiver acima do limite de pessoas. Para entrar pessoas a porta deve está aberta. O prédio tem 15 andares e o limite de pessoas é 10.

Código Principal

import java.util.Scanner; public class Principal { public static void main(String[] args) { ClasseEMetodo elevador = new ClasseEMetodo(); elevador.andarAtual=0; elevador.pessoasAtual=0; elevador.porta=1; elevador.pessoas=10; elevador.andares=15; Scanner scan = new Scanner(System.in); int menu=0; System.out.println("Status Atual do Elevador"); elevador.status(); while(menu!=7) { System.out.println(" "); System.out.println("Digite a açăo desejada"); System.out.println(" "); System.out.println("1 - Mover elevador"); System.out.println("2 - Abrir porta"); System.out.println("3 - Fechar porta"); System.out.println("4 - Entrar pessoas"); System.out.println("5 - Sair pessoas"); System.out.println("6 - Status Atual do elevador"); System.out.println("7 - Sair do programa"); menu = scan.nextInt(); if(menu==1) { if(elevador.porta==2) { System.out.println("Digite o andar que o elevador vai"); int valor = scan.nextInt(); elevador.moverElevadorPara(valor); }else { System.err.println("Movimentaçăo inválida, porta está aberta"); } } else if(menu==2) { elevador.abrirPorta(); } else if(menu==3) { elevador.fecharPorta(); } else if(menu==4) { System.out.println("Digite a quantidade de pessoas que irăo entrar"); int valor = scan.nextInt(); elevador.entrarPessoas(valor); } else if(menu==5) { if(elevador.porta==1 && elevador.pessoasAtual!=0) { System.out.println("Digite a quantidade de pessoas que irăo sair"); int valor = scan.nextInt(); elevador.sairPessoas(valor); }else if(elevador.porta==2){ System.err.println("Movimentaçăo inválida, porta está fechada"); }else if(elevador.pessoasAtual==0) { System.err.println("Movimentaçăo inválida, elevador está vázio"); } } else if(menu==6) { elevador.status(); } else if(menu==7) { System.out.println("Programa finalizado"); scan.close(); } else { System.err.println("Número inválido"); } } } }

Métodos

public class ClasseEMetodo { int pessoas=10, andares=15; int pessoasAtual=0, andarAtual=0; int porta=1; void moverElevadorPara(int valor) { if(andarAtual<=andares && porta==2 && valor<=15) { andarAtual = valor; System.out.println("Elevador agora está no: "+valor+" andar"); }else if(porta==1) { System.err.println("Movimentaçăo inválida, porta está aberta"); }else { System.err.println("Movimentaçăo inválida, prédio năo possui esse andar"); } } //se 1 porta aberta se 2 porta fechada void abrirPorta() { porta=1; } void fecharPorta() { porta=2; } void pessoasAtual() { System.out.println(pessoasAtual); } void entrarPessoas(int valor) { if(porta==1 && valor+pessoasAtual<=10) { pessoasAtual += valor; System.out.println("Elevador possui: "+valor+" nova(s) pessoa(s)"); }else if(porta==2) { System.err.println("Movimento inválido, porta está fechada"); }else { System.err.println("Limite de pessoas atingindo"); } } void sairPessoas(int valor) { if(porta==1 && pessoasAtual!=0 && valor<=pessoasAtual) { pessoasAtual -= valor; System.out.println("Saiu do elevador: "+valor+" pessoa(s)"); }else if(porta == 2){ System.err.println("Movimentaçăo inválida, porta fechada"); }else if(pessoasAtual==0){ System.err.println("Movimentaçăo inválida, elevador está vázio"); }else if(valor>pessoasAtual) { System.err.println("Năo existe tantas pessoas no elevador"); } } void status() { System.out.println(" "); System.out.println("O elevador tem: "+pessoasAtual+" pessoa(s)"); if(porta==1) { System.out.println("Porta: aberta"); }else { System.out.println("Porta: fechada"); } System.out.println("O andar atual é: "+andarAtual); } }
